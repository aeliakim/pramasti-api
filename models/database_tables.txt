
Table users {
  user_id int [primary key, increment]
  nrp int
  nama varchar(255)
  password varchar(255)
  email varchar(255)
  nohp varchar(20)
  departemen varchar(255)
  role enum('praktikan', 'asisten', 'koordinator', 'dosen', 'admin')
  status
  approved by varchar(255)
  last state
  //tambahkan kolom status, approved by, last state
  profil_picture varchar(255) //link ke file gambar
}

Table praktikum {
  praktikum_id int [primary key, increment]
  praktikum_name varchar(255)
  deskripsi text
  logo_praktikum varchar(255) //link logo praktikum
}

Table jadwalPraktikum {
  jadwal_id int [primary key, increment]
  praktikum_id int
  judul_modul varchar(255)
  tanggal date
  waktu_mulai time
  kuota int
}

Table kelompok {
  kelompok_id int [primary key, increment]
  jadwal_id int
  kapasitas int // jumlah maksimal anggota kelompok
  nama_kelompok varchar(255)
}

Table mhsPilihPraktikum {
  mahasiswa_id int
  praktikum_id int
  jadwal_id int
  kelompok_id int
}

Table asisten {
  asisten_id int [primary key]
  praktikum_id int
}

Table nilai {
  nilai_id int [primary key]
  mahasiswa_id int
  praktikum_id int
  nilai decimal(5,2)
}

Table pengumuman {
  pengumuman_id int [primary key]
  judul varchar(255)
  isi_pengumuman text
  tanggal date
  oleh_id int
}

Table asistenJadwal {
  asisten_id int
  jadwal_id int
}

Table roles {
  role_id int
  role_name varchar(100)
  user_id int
}
